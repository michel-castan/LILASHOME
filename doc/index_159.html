Transition ::= ( <ON> Expression )? <TARGET> Identificateur
Instruction ::= ( ( ( ArrayIdentificateur | Distribution | Identificateur | IdentificateurLocal | IdentificateurInternal ) "<=" Expression ) | ArrayDef | InstructionIf ) ";"
InstructionIf ::= <IF> Expression InstructionThen ( <ELSIF> Expression InstructionThen )* ( InstructionElse )? <END> <IF>
